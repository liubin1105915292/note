merge方法
MERGE INTO EIS_WISEYE_USER A
USING (SELECT B.USER_ID, B.ORG_ID FROM EIS_WISEYE_USER_TEMP B) C
ON (A.USER_ID = C.USER_ID AND C.USER_ID IN(323794066, 4531493490)) 
WHEN MATCHED THEN
  UPDATE SET A.ORG_ID = C.ORG_ID

  
  WHEN NOT MATCHED THEN
  INSERT(A.ID,A.NAME,A.YEAR) VALUES(C.AID,C.NAME,C.YEAR);
